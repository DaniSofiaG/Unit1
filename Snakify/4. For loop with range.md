# Series -2
```.py
#Given two integers A and B. Print all numbers from A to B inclusively, in ascending order, if A < B, or in descending order, if A â‰¥ B.
A = int(input())
B = int(input())

if A==B:
    print(A)
    
if B > A:
    for i in range(A,B+1):
        print(i, sep=', ', end=' ')

if A > B:
    for i in reversed(range(B,A+1)):
        print(i, sep=', ', end=' ')
 ```
